def extract_air_vector_from_motion(frames, start, end):
    try:
        if not mocap.joint_channels("Head") or "Xrotation" not in mocap.joint_channels("Head"):
            return None
        idx = mocap.get_joint_channel_index("Head", "Xrotation")
        segment = [float(frames[i][idx]) for i in range(start, end + 1)]
        if len(segment) < 2:
            return None
        delta = np.std(segment)
        tension = min(delta / 30, 1.0)
        flow = 1.0 - tension
        return {
            "tension": round(tension, 3),
            "ambiguity": 0.5,
            "flow": round(flow, 3),
            "density": 0.6,
            "resonance": 0.4,
            "freedom": 0.7
        }
    except Exception as e:
        print(f"⚠️ 空気ベクトル抽出エラー: {e}")
        return None

#計算機科学/Coding 
なおpythonを扱う場合には「py」スクリプトを使うべきね。じゃないと柔軟に使えないんで。

ほんでさ、myenvってのに入れといた方が良い。確定で。「```myenv\scripts\activate```」ってんで入れますんで。
ほんでpipが入ってないはずなんで環境変数か復旧かをすればよいんすけど。

まずは「```where pip```」で反応があるかどうかを見ますね。そんで反応あるならそれで良き。
反応ないならまずは環境変数確認。「Path」ってところを編集で見てみてほしい。

もし「```～～\python〇〇\Scripts```」ってのがあるんなら復旧ですー。
「```python -m ensurepip --upgrade```」で復旧できます。

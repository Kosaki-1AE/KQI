#Mine研究/ワイ理論 
0.5秒ごとに総合活性度を求めた状態で、それをグラフとして出力したら、そのときの空気感の向きというかが分かる気がする。でもそれには責任の矢というベクトルの向きが必要にはなると思うんだけどね。これが意思からの直接送られるデータになるんだけどさ。
以下、プログラムです。

import numpy as np
import matplotlib.pyplot as plt

###### ダミーデータ
t = np.arange(0, 10, 0.5)             # 0～9.5秒
A = np.random.rand(len(t)) * 2        # 活性度サンプル
theta = np.linspace(0, 2*np.pi, len(t))  # 責任の矢の角度サンプル

###### quiver 用にベクトル成分に分解
U = np.cos(theta)  # x成分
V = np.sin(theta)  # y成分

fig, ax = plt.subplots(figsize=(8,4))
###### 矢印を (t, A) の位置にプロット
ax.quiver(t, A, U, V, angles='xy', scale_units='xy', scale=1, width=0.005)
ax.set_xlabel("Time (s)")
ax.set_ylabel("Activation (A = ΔE + ΔE/Δt)")
ax.set_title("空気感の強度と責任の矢の向き")
plt.show()
